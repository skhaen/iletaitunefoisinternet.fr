Configurations

Bind - serveur faisant autorité

# Le plus simple serveur faisant autorité 
# pour le TLD "example"

options {
        recursion no;
};
zone "example" {
        type master;
        file "example";
};

Bind - résolveur

# Le plus simple résolveur
acl me {
                2001:db8:43::/48;
};
options {
        recursion yes;
        allow-recursion   { mynetwork; };
        allow-query-cache { mynetwork; };
        allow-query { mynetwork; };
};

Bind - données

@       IN      SOA     ns1.nic root@nic (
                2013071800              ; Serial
                7200            ; Refresh
                1800            ; Retry
                2419200         ; Expire
                600 )   ; Negative Cache TTL

@       IN      NS      ns1.nic.example.
        IN      NS      ns1.pch.net.

www     IN    AAAA   2001:db8::bad:dcaf

NSD

zone:         
        name: "example"
        zonefile: "example"

Unbound

server:
  interface: ::0
  interface: 0.0.0.0
  access-control: 2001:db8:43::/48 allow
